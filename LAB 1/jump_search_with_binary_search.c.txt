#include<stdio.h>
void search(int n,int key,int arr[])
{
    int l=0,h=n-1;
    if(arr[l]==key)
    {
        printf("key found!!!!"); 
    }
    else if(arr[h]==key)
    {
        printf("key found!!!!");
    }
    else if(arr[h]<key)
    {
        printf("key Not found!!!!");
    }
    else
    {
        l=1;
        int k;
        while(l<h)
        {
            if(arr[l]>=key)
            {
                break;
            }
            l=l*2;
        }
        if(l>h)
        {
            l=h;
        }
        l=k;
        int i, low, high, mid;
        low = 0;
        high = n - 1;
        mid = (low+high)/2;
        while (low <= high)
        {
            mid = (low+high)/2;
            if (arr[mid] == key)
            {
                printf("Key found!!!\n");
                break;
            }
            else if(arr[mid] < key)
            {
                low = mid+1;
            }
            else
            {
                high = mid-1;
            }
        }
        if(low > high)
        {
            printf("Key Not found!!!!!\n");
        }
    }
}
int main()
{
    int n,i,key;
    printf("Enter no. of elements: ");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements: ");
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("Enter the key value you want to search: ");
    scanf("%d",&key);
    search(n,key,arr);
}
