#include<stdio.h>
int search(int n,int key,int arr[])
{
    int flag=0;
    for(int i=0;i<n;i++)
    {
        if(arr[i]=key)
        {
            flag = 1;
            break;
        }
    }if(flag==1)
    {
        printf("KEY FOUND!!!!");
    }
    else 
    {
        printf("KEY NOT FOUND!!!!");
    }


}
int main()
{
    int n,i,key;
    printf("Enter no. of elements: ");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements: ");
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("Enter the key value you want to search: ");
    scanf("%d",&key);
    search(n,key,arr);
}
